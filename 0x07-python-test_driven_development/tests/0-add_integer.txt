#!/usr/bin/python3
import unittest

class TestAddInteger(unittest.TestCase):
    def test_add_integer(self):
        self.assertEqual(add_integer(5, 6), 11)
        self.assertEqual(add_integer(5.5, 6), 11)
        self.assertEqual(add_integer(5.5, 6.5), 11)
        
    def test_invalid_input(self):
        with self.assertRaises(TypeError):
            add_integer(5, "6")

if __name__ == '__main__':
    unittest.main()
